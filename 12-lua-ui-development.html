12 - Lua UI 开发实战

游戏 UI 是 Lua 最密集的应用场景之一。本章讲解用 Lua 驱动 UI 的常见架构模式、事件系统和生命周期管理——不是「如何在 Lua 里摆按钮」，而是「如何设计一个可维护的 Lua UI 框架」。

为什么用 Lua 写 UI

UI 是游戏开发中变更最频繁的系统——策划改一个按钮位置、调整一个文字颜色、新增一个弹窗，这些如果每次都要改 C# 重新编译，效率太低。Lua 写 UI 意味着：

  • 改 UI 逻辑不用重新编译 C#
  • 热更新可以直接修 UI bug
  • 策划可以在一定程度上调整 UI 行为

UI 框架核心：MVC 模式

在 Lua UI 中，最经典的架构是 MVC：

层职责Lua/C# 分工
Model数据 + 业务逻辑纯 Lua
ViewUI 控件、布局、动画C# 提供底层（UGUI），Lua 控制属性
Controller事件响应、Model-View 同步纯 Lua
 
-- Model：纯数据 + 变更通知
local PlayerModel = {
    name = "",
    level = 1,
    hp = 100,
    maxHp = 100,
    gold = 0,
    _listeners = {},  -- 观察者列表
}

function PlayerModel:set(key, value)
    if self[key] ~= value then
        local old = self[key]
        self[key] = value
        self:notify(key, old, value)
    end
end

function PlayerModel:on(key, callback)
    self._listeners[key] = self._listeners[key] or {}
    table.insert(self._listeners[key], callback)
end

function PlayerModel:notify(key, old, new)
    for _, cb in ipairs(self._listeners[key] or {}) do
        cb(old, new)
    end
end

-- View：绑定 UI 控件
local PlayerView = {}

function PlayerView:init(go)
    -- 通过 XLua 获取 UGUI 组件引用
    self.name_text = go:FindChild("NameText"):GetComponent("Text")
    self.hp_slider = go:FindChild("HPSlider"):GetComponent("Slider")
    self.gold_text = go:FindChild("GoldText"):GetComponent("Text")
    self.levelup_btn = go:FindChild("LevelUpBtn"):GetComponent("Button")
end

function PlayerView:bind(model)
    -- 数据变化时自动更新 UI
    model:on("name", function(_, new) self.name_text.text = new end)
    model:on("hp", function(_, new) self.hp_slider.value = new / model.maxHp end)
    model:on("gold", function(_, new) self.gold_text.text = tostring(new) end)
    model:on("level", function(_, new)
        -- 升级特效
        self:play_level_up_effect()
    end)

    -- 按钮事件 → Controller
    self.levelup_btn.onClick:AddListener(function()
        self.on_levelup_clicked()
    end)
end

-- Controller：协调 Model 和 View
local PlayerController = {
    model = nil,
    view = nil,
}

function PlayerController:init(model, view)
    self.model = model
    self.view = view
    self.view.on_levelup_clicked = function() self:handle_levelup() end
    view:bind(model)
end

function PlayerController:handle_levelup()
    if self.model.gold >= 100 then
        self.model:set("gold", self.model.gold - 100)
        self.model:set("level", self.model.level + 1)
        self.model:set("maxHp", self.model.maxHp + 20)
        self.model:set("hp", self.model.maxHp)
    end
end

UI 生命周期管理

每个 UI 面板都有明确的生命周期。管理不善会导致内存泄漏、事件残留、层级混乱。

-- UI 面板基类
local UIPanel = {}
UIPanel.__index = UIPanel

function UIPanel.new(template)
    local panel = setmetatable(template or {}, UIPanel)
    panel._events = {}   -- 事件注册记录
    panel._timers = {}   -- 定时器记录
    return panel
end

function UIPanel:on_create(go)
    -- 绑定 UI 引用
    self.gameObject = go
    self.transform = go.transform
end

function UIPanel:on_show(...)
    -- 面板显示：注册事件、启动动画
    self:register_events()
    self:play_show_animation()
end

function UIPanel:on_hide()
    -- 面板隐藏：注销事件、停止动画
    self:unregister_events()
    self:stop_all_timers()
end

function UIPanel:on_destroy()
    -- 面板销毁：释放所有资源
    self:unregister_events()
    self:stop_all_timers()
    self:release_ui_refs()
    self.gameObject = nil
end

function UIPanel:register_event(target, event, callback)
    target[event] = target[event] + callback  -- 简化示意
    table.insert(self._events, {target = target, event = event, callback = callback})
end

function UIPanel:unregister_events()
    for _, evt in ipairs(self._events) do
        evt.target[evt.event] = evt.target[evt.event] - evt.callback
    end
    self._events = {}
end

function UIPanel:delay(seconds, callback)
    local timer = {time = seconds, callback = callback}
    table.insert(self._timers, timer)
end

function UIPanel:stop_all_timers()
    self._timers = {}
end

function UIPanel:release_ui_refs()
    -- 释放所有 C# 引用（XLua 的 Dispose 或置 nil）
    for k, _ in pairs(self) do
        if type(self[k]) == "userdata" then
            self[k] = nil  -- 让 GC 回收
        end
    end
end

-- 使用示例
local ShopPanel = UIPanel.new({
    name = "ShopPanel",
})

function ShopPanel:on_create(go)
    UIPanel.on_create(self, go)
    -- 获取 UI 引用...
end

function ShopPanel:on_show(...)
    UIPanel.on_show(self, ...)
    -- 加载商品数据，刷新列表...
end

事件总线：跨面板通信

面板之间不应该直接引用对方——用事件总线解耦。

-- 全局事件总线
local EventBus = {
    _handlers = {},
}

function EventBus:on(event, handler)
    self._handlers[event] = self._handlers[event] or {}
    table.insert(self._handlers[event], handler)
end

function EventBus:off(event, handler)
    local list = self._handlers[event]
    if not list then return end
    for i = #list, 1, -1 do
        if list[i] == handler then
            table.remove(list, i)
        end
    end
end

function EventBus:emit(event, ...)
    for _, handler in ipairs(self._handlers[event] or {}) do
        handler(...)
    end
end

-- 面板间通信示例
-- 背包面板：使用物品
EventBus:emit("ITEM_USED", itemId)

-- 角色面板：监听物品使用
EventBus:on("ITEM_USED", function(itemId)
    -- 刷新角色属性显示
    self:refresh_stats()
end)

-- 任务面板：监听物品使用
EventBus:on("ITEM_USED", function(itemId)
    -- 检查任务进度
    self:check_quest_progress(itemId)
end)

-- 面板关闭时注销
function ShopPanel:on_hide()
    UIPanel.on_hide(self)
    EventBus:off("ITEM_USED", self.on_item_used)
end

属性变更通知：从 Model 到 UI 的桥梁

-- 可观察属性（Observable）
-- 类似 Vue 的响应式——数据变了，UI 自动更新

local Observable = {}

function Observable.new(initial)
    local obj = {
        _value = initial,
        _watchers = {},
    }
    setmetatable(obj, Observable)
    return obj
end

function Observable:get()
    return self._value
end

function Observable:set(new_value)
    if self._value ~= new_value then
        local old = self._value
        self._value = new_value
        for _, watcher in ipairs(self._watchers) do
            watcher(new_value, old)
        end
    end
end

function Observable:watch(callback)
    table.insert(self._watchers, callback)
    -- 返回注销函数
    return function()
        for i, cb in ipairs(self._watchers) do
            if cb == callback then
                table.remove(self._watchers, i)
                return
            end
        end
    end
end

-- 使用：数据和 UI 的自动同步
local playerGold = Observable.new(0)

-- UI 绑定
local unwatch = playerGold:watch(function(newGold, oldGold)
    goldText.text = tostring(newGold)
    if newGold > oldGold then
        play_gold_effect()  -- 获得金币特效
    end
end)

-- 业务逻辑
playerGold:set(100)   -- UI 自动更新
playerGold:set(150)   -- UI 自动更新 + 播放特效

-- 不再需要时解绑
unwatch()

列表滚动优化

-- 游戏中最常见的 UI 性能问题：长列表
-- 解决方案：虚拟列表（只创建可见区域的 item）

local VirtualList = {}

function VirtualList.new(scroll_view, item_prefab)
    local list = {
        scroll = scroll_view,
        prefab = item_prefab,
        item_height = 100,       -- 每个 item 的高度
        buffer = 3,              -- 上下缓冲数量
        data = {},
        items = {},              -- 当前可见的 item 列表
        first_visible = 1,
    }
    setmetatable(list, {__index = VirtualList})
    return list
end

function VirtualList:set_data(data_list)
    self.data = data_list
    self:_refresh()
end

function VirtualList:on_scroll(scroll_pos)
    local new_first = math.floor(scroll_pos / self.item_height) + 1
    if new_first ~= self.first_visible then
        self.first_visible = new_first
        self:_refresh()
    end
end

function VirtualList:_refresh()
    -- 回收不可见的 item 到对象池
    self:_recycle_invisible()
    -- 计算可见范围
    local visible_count = math.ceil(self.scroll.viewport_height / self.item_height)
    local start = math.max(1, self.first_visible - self.buffer)
    local finish = math.min(#self.data, start + visible_count + self.buffer * 2)
    -- 创建/复用可见 item
    for i = start, finish do
        if not self.items[i] then
            self.items[i] = self:_get_item()
        end
        self:_update_item(self.items[i], self.data[i], i)
    end
end

-- 1000 条数据，只创建 ~15 个 item 对象
-- 性能从 O(n) 变成 O(1)
虚拟列表 vs 简单 ScrollView：如果你把 1000 个 item 全部创建出来，光是 GameObject 的实例化就够跑几秒钟了。虚拟列表始终只维护可见区域 + 缓冲区内的 item（通常 15-20 个），列表滚到哪创建到哪。

练习与思考

1

基础练习

实现一个 Observable 的 computed 功能——一个值依赖于其他 Observable，当依赖变化时自动重新计算（类似 Vue 的 computed）。

2

架构练习

设计一个完整的背包 UI 系统（Model + View + Controller），包含物品列表、拖拽排序、物品使用/丢弃。画出各层之间的数据流。

3

性能优化

实现虚拟列表中 item 的对象池。池满时回收最久未用的 item，池空时创建新的。测量有对象池和无对象池的 GC 内存分配差异。

本章小结

  • MVC 模式：Model（数据+通知）、View（UI 绑定）、Controller（协调）
  • UI 生命周期：on_create → on_show → on_hide → on_destroy，每阶段做对应的事
  • 事件总线解耦面板间通信，面板关闭时必须注销事件
  • Observable 实现数据驱动的 UI 更新，减少手动刷新代码
  • 虚拟列表让长列表从 O(n) 降到 O(1)，是列表性能的必选方案
