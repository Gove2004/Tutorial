实战项目：Lua 背包系统

目标：构建一个游戏背包系统，包含物品存储、堆叠、排序、使用、位置交换。纯 Lua 实现（不依赖 XLua/Unity），可以直接在命令行运行。

-- item.lua
local Item = {}
Item.__index = Item

function Item.new(data)
    local item = setmetatable({}, Item)
    item.id = data.id
    item.name = data.name or "Unknown"
    item.type = data.type or "misc"
    item.max_stack = data.max_stack or 99
    item.description = data.description or ""
    item.usable = data.usable or false
    item.use_effect = data.use_effect  -- 使用时的效果函数
    return item
end

function Item:can_stack_with(other)
    return self.id == other.id and self.max_stack > 1
end

function Item:use(target)
    if self.usable and self.use_effect then
        return self.use_effect(target)
    end
    return false, "Cannot use this item"
end

-- slot.lua
local Slot = {}
Slot.__index = Slot

function Slot.new(index)
    return setmetatable({
        index = index,
        item = nil,
        count = 0,
    }, Slot)
end

function Slot:is_empty()
    return self.item == nil or self.count <= 0
end

function Slot:can_accept(item, count)
    if self:is_empty() then
        return true
    end
    return self.item:can_stack_with(item)
        and (self.count + count) <= self.item.max_stack
end

function Slot:add(item, count)
    count = count or 1
    if self:is_empty() then
        self.item = item
        self.count = count
    else
        self.count = self.count + count
    end
end

function Slot:remove(count)
    count = count or self.count
    self.count = self.count - count
    if self.count <= 0 then
        self.item = nil
        self.count = 0
    end
    return count
end

function Slot:clear()
    self.item = nil
    self.count = 0
end

-- event.lua
local EventBus = {}

local handlers = {}

function EventBus.on(event, callback)
    handlers[event] = handlers[event] or {}
    table.insert(handlers[event], callback)
end

function EventBus.emit(event, ...)
    for _, cb in ipairs(handlers[event] or {}) do
        cb(...)
    end
end

function EventBus.off(event, callback)
    local list = handlers[event]
    if not list then return end
    for i = #list, 1, -1 do
        if list[i] == callback then
            table.remove(list, i)
        end
    end
end

-- backpack.lua
local Backpack = {}
Backpack.__index = Backpack

function Backpack.new(capacity)
    local bp = setmetatable({
        capacity = capacity or 30,
        slots = {},
        _item_index = {},  -- itemId → slot 索引表，快速查找
    }, Backpack)

    -- 预创建所有 slot
    for i = 1, bp.capacity do
        bp.slots[i] = Slot.new(i)
    end

    return bp
end

-- 添加物品（核心方法）
function Backpack:add_item(item, count)
    count = count or 1
    local remaining = count

    -- 先尝试堆叠到已有的同类 slot
    if item.max_stack > 1 then
        for _, slot in ipairs(self.slots) do
            if not slot:is_empty() and slot.item.id == item.id then
                local space = item.max_stack - slot.count
                local to_add = math.min(space, remaining)
                if to_add > 0 then
                    slot:add(item, to_add)
                    remaining = remaining - to_add
                end
                if remaining <= 0 then break end
            end
        end
    end

    -- 剩余放到空 slot
    while remaining > 0 do
        local slot = self:find_empty_slot()
        if not slot then
            EventBus.emit("BACKPACK_FULL", self)
            return false, count - remaining
        end
        local to_add = math.min(item.max_stack, remaining)
        slot:add(item, to_add)
        remaining = remaining - to_add
    end

    EventBus.emit("BACKPACK_CHANGED", self)
    return true, count
end

-- 交换两个 slot（拖拽排序）
function Backpack:swap_slots(idx_a, idx_b)
    local slot_a = self.slots[idx_a]
    local slot_b = self.slots[idx_b]
    if not slot_a or not slot_b then return false end

    -- 如果可以堆叠，则合并
    if not slot_a:is_empty() and not slot_b:is_empty()
        and slot_a.item.id == slot_b.item.id
        and slot_a.item.max_stack > 1 then
        local total = slot_a.count + slot_b.count
        local maxstack = slot_a.item.max_stack
        slot_a.count = math.min(total, maxstack)
        slot_b.count = total - slot_a.count
        if slot_b.count <= 0 then slot_b:clear() end
    else
        -- 直接交换
        slot_a.item, slot_b.item = slot_b.item, slot_a.item
        slot_a.count, slot_b.count = slot_b.count, slot_a.count
    end

    EventBus.emit("BACKPACK_CHANGED", self)
    return true
end

-- 快速查找物品
function Backpack:has_item(item_id, min_count)
    min_count = min_count or 1
    local total = 0
    for _, slot in ipairs(self.slots) do
        if not slot:is_empty() and slot.item.id == item_id then
            total = total + slot.count
            if total >= min_count then return true end
        end
    end
    return false
end

-- 移除物品
function Backpack:remove_item(item_id, count)
    count = count or 1
    local removed = 0
    -- 从前往后遍历，优先移除数量最少的 slot
    for _, slot in ipairs(self.slots) do
        if not slot:is_empty() and slot.item.id == item_id then
            local to_remove = math.min(slot.count, count - removed)
            slot:remove(to_remove)
            removed = removed + to_remove
            if removed >= count then break end
        end
    end
    EventBus.emit("BACKPACK_CHANGED", self)
    return removed
end

-- 按类型排序
function Backpack:sort_by_type()
    table.sort(self.slots, function(a, b)
        if a:is_empty() and b:is_empty() then return false end
        if a:is_empty() then return false end
        if b:is_empty() then return true end
        return a.item.type < b.item.type
    end)
    -- 更新 index
    for i, slot in ipairs(self.slots) do
        slot.index = i
    end
    EventBus.emit("BACKPACK_SORTED", self)
end

function Backpack:find_empty_slot()
    for _, slot in ipairs(self.slots) do
        if slot:is_empty() then return slot end
    end
    return nil
end

function Backpack:get_used_count()
    local count = 0
    for _, slot in ipairs(self.slots) do
        if not slot:is_empty() then count = count + 1 end
    end
    return count
end

function Backpack:to_table()
    -- 序列化为简单 table（存档用）
    local t = {capacity = self.capacity, items = {}}
    for _, slot in ipairs(self.slots) do
        if not slot:is_empty() then
            table.insert(t.items, {
                id = slot.item.id,
                count = slot.count,
                index = slot.index,
            })
        end
    end
    return t
end

function Backpack:from_table(data)
    -- 从序列化数据恢复（读档用）
    self.capacity = data.capacity
    self.slots = {}

        self.slots[i] = Slot.new(i)

end

-- main.lua（测试用例）
local function test_backpack()
    local bp = Backpack.new(20)

    -- 创建物品
    local health_potion = Item.new({
        id = "health_potion",
        name = "生命药水",
        type = "consumable",
        max_stack = 99,
        usable = true,
        use_effect = function(target)
            target.hp = math.min(target.hp + 50, target.maxHp)
            return true
        end,
    })

    local sword = Item.new({
        id = "iron_sword",
        name = "铁剑",
        type = "weapon",
        max_stack = 1,
    })

    local shield = Item.new({
        id = "wooden_shield",
        name = "木盾",
        type = "armor",
        max_stack = 1,
    })

    -- 测试 1：添加物品
    bp:add_item(health_potion, 5)
    bp:add_item(sword, 1)
    bp:add_item(shield, 1)
    assert(bp:get_used_count() == 3)

    -- 测试 2：堆叠
    bp:add_item(health_potion, 3)
    -- 第一个有药水的 slot 应该变成 8
    local potion_slot = nil
    for _, slot in ipairs(bp.slots) do
        if not slot:is_empty() and slot.item.id == "health_potion" then
            potion_slot = slot
            break
        end
    end
    assert(potion_slot.count == 8)

    -- 测试 3：查找
    assert(bp:has_item("health_potion", 5))
    assert(not bp:has_item("gold_bar"))

    -- 测试 4：移除
    bp:remove_item("health_potion", 3)
    -- potion_slot 现在应该是 5
    assert(potion_slot.count == 5)

    -- 测试 5：交换
    local sword_idx = nil
    for i, slot in ipairs(bp.slots) do
        if not slot:is_empty() and slot.item.id == "iron_sword" then
            sword_idx = i
            break
        end
    end
    local shield_idx = nil
    for i, slot in ipairs(bp.slots) do
        if not slot:is_empty() and slot.item.id == "wooden_shield" then
            shield_idx = i
            break
        end
    end
    bp:swap_slots(sword_idx, shield_idx)
    assert(bp.slots[sword_idx].item.id == "wooden_shield")
    assert(bp.slots[shield_idx].item.id == "iron_sword")

    print("All tests passed!")
end

    result = result .. "a"
    -- 每次 .. 都创建新字符串！
end
 
local parts = {}
for i = 1, 10000 do
    parts[i] = "a"
end
local result = table.concat(parts)

性能检查清单

上线前必查：以下每一项如果没做到，都可能成为线上事故。
  1. 所有变量都是 local 吗？——遍历整个项目，全局变量 grep 一遍。漏一个全局就是一颗炸弹
  2. 循环中在做字符串拼接吗？——用 table.concat 替换
  3. table 有空洞吗？——如果有，# 的行为不可预测
  4. 大量的临时 table 在循环/Update 中创建吗？——用对象池或 table.create 预分配
  5. C# ↔ Lua 的跨语言调用在热路径上吗？——XLua 项目中，每帧的跨语言调用总开销应 < 1ms
  6. LuaFunction/LuaTable Dispose 了吗？——XLua 项目中忘记 Dispose = GC 泄漏
  7. 闭包持有大对象 upvalue 吗？——不需要的 upvalue 置 nil
  8. 协程对象释放了吗？——suspended 状态的协程持有其栈上所有对象
  9. 事件回调在不再需要时注销了吗？——UI 销毁时事件残留 = 内存泄漏
  10. 弱引用表对缓存/监听模式用了吗？——防止缓存让对象永不被回收

面试高频陷阱

陷阱 1：Lua 中 0 是 true 还是 false？

true。Lua 只有 nil 和 false 为假。0、空字符串、空 table 全部为真。面试官最爱拿这个考 C/C++ 转 Lua 的人。

陷阱 2：Lua 数组从 0 还是 1 开始？为什么？

从 1 开始。原因是 Lua 的 table 不是真正的数组——1 是最自然的第一个正整数键。标准库函数（table.insert、table.remove、ipairs、# 对数组部分的语义）都以此为约定。但 t[0] 可以用（进哈希部分）。

陷阱 3：ipairs 和 pairs 有什么区别？什么时候用哪个？

ipairs 按整数键从小到大迭代数组部分，遇到 nil 就停止。pairs 迭代所有键值对，顺序不保证。遍历已知的数组用 ipairs，遍历字典/混合 table 用 pairs。

陷阱 4：local a, b = f() 和 local a = f() 接收多返回值的区别？

local a, b = f() 收 f() 返回的第一个和第二个值。local a = f() 只收第一个值，其余丢弃。多返回值只有当函数调用在表达式末尾时才全部保留。

陷阱 5：Lua 的 GC 是什么类型？5.4 有什么改进？

5.1-5.3：增量式三色标记-清除。5.4：引入分代 GC（generational），minor GC 只扫新生代 + 跨代引用，major GC 全扫。分代 GC 对「大量临时对象用完即弃」的场景（如游戏每帧）性能提升明显。

陷阱 6：# 运算符在什么情况下结果是不可预测的？

当 table 的数组部分有空洞（中间有 nil）时，# 的行为未定义。可能返回空洞前的长度、空洞后的长度、或任何值。只有紧密数组（无空洞）才能依赖 #。

陷阱 7：Lua 如何实现 OOP？没有 class 关键字怎么办？

用 table + 元表。table 存储实例数据和方法，__index 元方法指向原型/类 table 实现继承。创建实例时 setmetatable({}, Class)。obj:method() 冒号语法糖自动传 self。

陷阱 8：为什么 XLua 需要「生成代码」？不生成会怎样？

XLua 默认用反射访问 C#，有大量 GC 分配和性能开销。生成代码预构建适配层，直接绑定委托，性能提升 10-100 倍。不生成代码时编辑器能跑但真机可能卡死。
